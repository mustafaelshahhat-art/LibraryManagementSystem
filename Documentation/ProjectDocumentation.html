1. Introduction

The Library Management System is a comprehensive desktop application developed to streamline and automate library operations. Built using C# and .NET 6 with Windows Forms, the system provides an efficient platform for managing books, authors, publishers, categories, members, and loan transactions.

This system replaces traditional manual record-keeping methods with a digital solution that ensures accuracy, speed, and reliability.

System Objectives

  • Automation: Eliminate manual logging and reduce human error
  • Efficiency: Speed up borrowing and return processes
  • Accuracy: Maintain accurate inventory and member records
  • Accessibility: Provide real-time book availability information
  • Security: Implement proper data validation and integrity constraints

3. ERD and Mapping

Figure 1: Entity Relationship Diagram

Database Schema - Part 1

1. PUBLISHER Table

CREATE TABLE PUBLISHER (
    Publisher_ID INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT NOT NULL,
    Address TEXT,
    Phone TEXT
);

2. AUTHOR Table

CREATE TABLE AUTHOR (
    Author_ID INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT NOT NULL,
    Bio TEXT,
    BirthDate TEXT
);

3. CATEGORY Table

CREATE TABLE CATEGORY (
    Category_ID INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT NOT NULL UNIQUE,
    Description TEXT
);

Database Schema - Part 2

4. BOOK Table

CREATE TABLE BOOK (
    ISBN TEXT PRIMARY KEY,
    Title TEXT NOT NULL,
    Year INTEGER,
    Publisher_ID INTEGER,
    FOREIGN KEY (Publisher_ID) REFERENCES PUBLISHER(Publisher_ID)
);

5. BOOK_AUTHOR Table

CREATE TABLE BOOK_AUTHOR (
    Book_ISBN TEXT,
    Author_ID INTEGER,
    PRIMARY KEY (Book_ISBN, Author_ID),
    FOREIGN KEY (Book_ISBN) REFERENCES BOOK(ISBN),
    FOREIGN KEY (Author_ID) REFERENCES AUTHOR(Author_ID)
);

6. BOOK_CATEGORY Table

CREATE TABLE BOOK_CATEGORY (
    Book_ISBN TEXT,
    Category_ID INTEGER,
    PRIMARY KEY (Book_ISBN, Category_ID),
    FOREIGN KEY (Book_ISBN) REFERENCES BOOK(ISBN),
    FOREIGN KEY (Category_ID) REFERENCES CATEGORY(Category_ID)
);

Database Schema - Part 3

7. BOOK_COPY Table

CREATE TABLE BOOK_COPY (
    Copy_ID INTEGER PRIMARY KEY AUTOINCREMENT,
    Book_ISBN TEXT,
    Status TEXT CHECK(Status IN ('Available', 'Loaned', 'Lost')) DEFAULT 'Available',
    FOREIGN KEY (Book_ISBN) REFERENCES BOOK(ISBN)
);

8. MEMBER Table

CREATE TABLE MEMBER (
    Member_ID INTEGER PRIMARY KEY AUTOINCREMENT,
    FirstName TEXT NOT NULL,
    LastName TEXT NOT NULL,
    Email TEXT,
    Phone TEXT,
    JoinDate TEXT DEFAULT CURRENT_DATE
);

9. LOAN Table

CREATE TABLE LOAN (
    Loan_ID INTEGER PRIMARY KEY AUTOINCREMENT,
    Book_Copy_ID INTEGER,
    Member_ID INTEGER,
    Loan_Date TEXT DEFAULT CURRENT_DATE,
    Due_Date TEXT,
    Return_Date TEXT,
    Status TEXT CHECK(Status IN ('Active', 'Returned', 'Overdue')) DEFAULT 'Active',
    FOREIGN KEY (Book_Copy_ID) REFERENCES BOOK_COPY(Copy_ID),
    FOREIGN KEY (Member_ID) REFERENCES MEMBER(Member_ID)
);
